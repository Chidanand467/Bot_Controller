class FaceTracker:
  def __init__(self, detector: FaceDetector, robot: RobotClient):
    # DEPENDENCY INJECTION for loose coupling
    self.detector = detector
    self.robot = robot
  def track_face(self, frame):
    face = self.detector.detect_largest_face(frame)
    if face:
      command = self._calculate_command(face)
      self.robot.send_command(command)
