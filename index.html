class RobotController:
    def __init__(self, detector: FaceDetector, navigator: NavigationSystem):
        # Dependency Injection for loose coupling
        self.detector = detector
        self.navigator = navigator
        self._status = "IDLE"
    
    def start_tracking(self):
        self._status = "TRACKING"
        # Start face detection thread
        Thread(target=self._tracking_loop).start()
    
    def _tracking_loop(self):
        while self._status == "TRACKING":
            frame = self.camera.capture_frame()
            faces = self.detector.detect(frame)
            if faces:
                command = self.navigator.calculate_command(faces[0])
                self.motor_controller.execute(command)
